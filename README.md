git init                           # Инициализация Git в папке  
git remote add origin https://github.com/laura))/my-website.git  # Подключаем репозиторий  
git add .                          # Добавляем все файлы  
git commit -m "Первая загрузка"    # Фиксируем изменения  
git branch -M main                 # Переименовываем ветку в main  
git push -u origin main            # Отправляем файлы в репозиторий
